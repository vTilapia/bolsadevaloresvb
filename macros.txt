Sub ExecutarOrdens

Dim ordens
Dim alunos
Dim carteira
Dim mercado

Dim i As Integer
Dim linhaAluno As Integer
Dim linhaMercado As Integer
Dim colunaEmpresa As Integer
Dim linhaInvestidor As Integer ' <--- Nova variável para alinhar a aba INVESTIDORES

Dim conta As String
Dim codigo As String
Dim tipo As String
Dim qtd As Double

Dim saldo As Double
Dim preco As Double
Dim valor As Double
Dim acoes As Double

ordens=ThisComponent.Sheets.getByName("ORDENS")
alunos=ThisComponent.Sheets.getByName("INVESTIDORES")
carteira=ThisComponent.Sheets.getByName("CARTEIRAS")
mercado=ThisComponent.Sheets.getByName("BOLSA DE VALORES")


For i=1 To 30

conta=ordens.getCellByPosition(1,i).String
tipo=UCase(ordens.getCellByPosition(3,i).String)
qtd=Val(Trim(ordens.getCellByPosition(4, i).String))
codigo=ordens.getCellByPosition(2,i).String

If conta<>"" Then

linhaAluno=-1

For j=1 To 100

If carteira.getCellByPosition(0,j).String=conta Then
linhaAluno=j
Exit For
End If

Next j


If linhaAluno=-1 Then
MsgBox("Conta "& conta &" não encontrada")
GoTo ProximaOrdem
End If


linhaMercado=-1

For k=1 To 100

If mercado.getCellByPosition(1,k).String=codigo Then
linhaMercado=k
Exit For
End If

Next k


If linhaMercado=-1 Then
MsgBox("Empresa "& codigo &" não encontrada")
GoTo ProximaOrdem
End If


colunaEmpresa=-1

For j=1 To 50

If carteira.getCellByPosition(j,0).String=codigo Then
colunaEmpresa=j
Exit For
End If

Next j

' =========================================================================
' AQUI ESTÁ A ADEQUAÇÃO DE LINHAS:
' No LibreOffice Basic, a Linha 2 da planilha é o índice 1 no código.
' Se linhaAluno na CARTEIRAS é 2 (índice 2), na INVESTIDORES deve ser a linha 2 (índice 1).
' Fórmula matemática de conversão de índices baseada em zero (0):
' =========================================================================
linhaInvestidor = (linhaAluno / 2)

' Agora usamos a linhaInvestidor correta para ler o saldo na aba INVESTIDORES
saldo=alunos.getCellByPosition(5, linhaInvestidor).Value

preco=mercado.getCellByPosition(5,linhaMercado).Value

valor=preco*qtd

acoes=carteira.getCellByPosition(colunaEmpresa,linhaAluno).Value


If tipo="COMPRA" Then

If saldo>=valor Then

saldo=saldo-valor
acoes=acoes+qtd

Else

MsgBox("Saldo insuficiente: "&conta)

End If

ElseIf tipo="VENDA" Then

If acoes>=qtd Then

saldo=saldo+valor
acoes=acoes-qtd

Else

MsgBox("Ações insuficientes: "&conta)

End If

End If

' Usamos a linhaInvestidor para salvar o novo saldo no aluno correto
alunos.getCellByPosition(5,linhaInvestidor).Value=saldo

carteira.getCellByPosition(colunaEmpresa,linhaAluno).Value=acoes


ProximaOrdem:

End If

Next i

MsgBox("Ordens executadas!")

End Sub

Sub GerarHtmlMercado
    Dim oDoc As Object
    Dim oSheet As Object
    Dim sUrl As String, sPath As String
    Dim i As Integer, barraPos As Integer
    Dim caractere As String
    Dim sHtmlPath As String
    
    Dim r As Integer
    Dim empresaNome As String
    Dim codigoAtivo As String
    Dim precoTexto As String
    Dim variacaoTexto As String
    Dim classeVariacao As String
    Dim setaVariacao As String
    Dim tabelaHtml As String
    Dim htmlCompleto As String
    
    oDoc = ThisComponent
    ' Ajuste o índice da aba (0 para a primeira aba, 1 para a segunda, etc.) onde está o mercado
    oSheet = oDoc.Sheets(5) 
    
    ' Descobre a pasta atual para salvar o mercado.html
    If oDoc.hasLocation() Then
        sUrl = oDoc.getLocation()
        sPath = ConvertFromUrl(sUrl)
        barraPos = 0
        For i = Len(sPath) To 1 Step -1
            caractere = Mid(sPath, i, 1)
            If caractere = "\" Or caractere = "/" Then
                barraPos = i
                Exit For
            End If
        Next i
        If barraPos > 0 Then
            sHtmlPath = Left(sPath, barraPos) & "mercado.html"
        Else
            sHtmlPath = sPath & ".html"
        End If
    Else
        MsgBox "Por favor, salve a sua planilha antes de gerar o HTML!", 48, "Erro"
        Exit Sub
    End If
    
    tabelaHtml = ""
    r = 1 ' Começa na linha 2 (índice 1) para pular o cabeçalho
    
    ' Varre as linhas da tabela até encontrar uma célula vazia
    Do
        empresaNome = Trim(oSheet.getCellByPosition(0, r).String)
        If empresaNome = "" Then Exit Do
        
        codigoAtivo  = Trim(oSheet.getCellByPosition(1, r).String)
        precoTexto   = Trim(oSheet.getCellByPosition(2, r).String)
        variacaoTexto = Trim(oSheet.getCellByPosition(3, r).String)
        
        ' Lógica para definir a cor e a setinha da variação (Alta ou Baixa)
        If InStr(variacaoTexto, "-") > 0 Then
            classeVariacao = "baixa"
            setaVariacao = "▼ "
        Else
            classeVariacao = "alta"
            setaVariacao = "▲ "
        End If
        
        ' Remove hífens ou caracteres estranhos que possam quebrar o visual
        variacaoTexto = Replace(variacaoTexto, "-", "")
        variacaoTexto = Replace(variacaoTexto, "?", "")
        
        ' Monta a linha da tabela em HTML limpo
        tabelaHtml = tabelaHtml & "            <tr>" & Chr(13)
        tabelaHtml = tabelaHtml & "                <td>" & empresaNome & "</td>" & Chr(13)
        tabelaHtml = tabelaHtml & "                <td>" & codigoAtivo & "</td>" & Chr(13)
        tabelaHtml = tabelaHtml & "                <td>" & precoTexto & "</td>" & Chr(13)
        tabelaHtml = tabelaHtml & "                <td class=""" & classeVariacao & """>" & setaVariacao & variacaoTexto & "</td>" & Chr(13)
        tabelaHtml = tabelaHtml & "            </tr>" & Chr(13)
        
        r = r + 1
    Loop
    
    ' MONTAGEM DO HTML (Fundo Escuro Fixo e sem tags duplicadas)
    htmlCompleto = "<!DOCTYPE html>" & Chr(13)
    htmlCompleto = htmlCompleto & "<html lang=""pt-BR"">" & Chr(13)
    htmlCompleto = htmlCompleto & "<head>" & Chr(13)
    htmlCompleto = htmlCompleto & "    <meta charset=""UTF-8"">" & Chr(13)
    htmlCompleto = htmlCompleto & "    <meta name=""viewport"" content=""width=device-width, initial-scale=1.0"">" & Chr(13)
    htmlCompleto = htmlCompleto & "    <style>" & Chr(13)
    htmlCompleto = htmlCompleto & "        body { margin: 0; padding: 10px; background: #0f172a; font-family: Arial, sans-serif; color: white; }" & Chr(13)
    htmlCompleto = htmlCompleto & "        h1 { text-align: center; font-size: 22px; margin-bottom: 15px; color: #ffffff; }" & Chr(13)
    htmlCompleto = htmlCompleto & "        .container { overflow-x: auto; max-width: 600px; margin: 0 auto; }" & Chr(13)
    htmlCompleto = htmlCompleto & "        table { width: 100%; border-collapse: collapse; background: #1e293b; border-radius: 12px; overflow: hidden; font-size: 14px; }" & Chr(13)
    htmlCompleto = htmlCompleto & "        th { background: #f4b400; color: black; padding: 12px 6px; font-weight: bold; }" & Chr(13)
    htmlCompleto = htmlCompleto & "        td { padding: 12px 6px; text-align: center; border-bottom: 1px solid #374151; color: white; }" & Chr(13)
    htmlCompleto = htmlCompleto & "        tr:hover { background: #374151; }" & Chr(13)
    htmlCompleto = htmlCompleto & "        .alta { color: #22c55e; font-weight: bold; }" & Chr(13)
    htmlCompleto = htmlCompleto & "        .baixa { color: #ef4444; font-weight: bold; }" & Chr(13)
    htmlCompleto = htmlCompleto & "        @media screen and (max-width:600px) { h1 { font-size: 18px; } table { font-size: 12px; } th, td { padding: 8px 4px; } }" & Chr(13)
    htmlCompleto = htmlCompleto & "    </style>" & Chr(13)
    htmlCompleto = htmlCompleto & "</head>" & Chr(13)
    htmlCompleto = htmlCompleto & "<body>" & Chr(13)
    htmlCompleto = htmlCompleto & "    <div class=""container"">" & Chr(13)
    htmlCompleto = htmlCompleto & "        <h1>Bolsa Escolar</h1>" & Chr(13)
    htmlCompleto = htmlCompleto & "        <table>" & Chr(13)
    htmlCompleto = htmlCompleto & "            <thead>" & Chr(13)
    htmlCompleto = htmlCompleto & "                <tr>" & Chr(13)
    htmlCompleto = htmlCompleto & "                    <th>Empresa</th>" & Chr(13)
    htmlCompleto = htmlCompleto & "                    <th>Código</th>" & Chr(13)
    htmlCompleto = htmlCompleto & "                    <th>Preço</th>" & Chr(13)
    htmlCompleto = htmlCompleto & "                    <th>Variação</th>" & Chr(13)
    htmlCompleto = htmlCompleto & "                </tr>" & Chr(13)
    htmlCompleto = htmlCompleto & "            </thead>" & Chr(13)
    htmlCompleto = htmlCompleto & "            <tbody>" & Chr(13)
    
    ' Injeta as linhas tratadas da tabela
    htmlCompleto = htmlCompleto & tabelaHtml
    
    htmlCompleto = htmlCompleto & "            </tbody>" & Chr(13)
    htmlCompleto = htmlCompleto & "        </table>" & Chr(13)
    htmlCompleto = htmlCompleto & "    </div>" & Chr(13)
    
    ' NOVO: INJEÇÃO AUTOMÁTICA DO SCRIPT COMPATÍVEL COM O INDEX.HTML
    htmlCompleto = htmlCompleto & "    <script>" & Chr(13)
    htmlCompleto = htmlCompleto & "        function lerTabelaESalvar() {" & Chr(13)
    htmlCompleto = htmlCompleto & "            let acoes = [];" & Chr(13)
    htmlCompleto = htmlCompleto & "            let linhas = document.querySelectorAll(""tbody tr"");" & Chr(13)
    htmlCompleto = htmlCompleto & "            linhas.forEach(linha => {" & Chr(13)
    htmlCompleto = htmlCompleto & "                let colunas = linha.querySelectorAll(""td"");" & Chr(13)
    htmlCompleto = htmlCompleto & "                if (colunas.length > 0) {" & Chr(13)
    htmlCompleto = htmlCompleto & "                    let ticker = colunas[1].innerText.trim();" & Chr(13)
    htmlCompleto = htmlCompleto & "                    let textoVariacao = colunas[3].innerText.trim();" & Chr(13)
    htmlCompleto = htmlCompleto & "                    let sinal = textoVariacao.includes(""▼"") ? -1 : 1;" & Chr(13)
    htmlCompleto = htmlCompleto & "                    let numeroLimpo = textoVariacao.replace(""▲"", """").replace(""▼"", """").replace(""%"", """").replace("","", ""."").trim();" & Chr(13)
    htmlCompleto = htmlCompleto & "                    let variacaoNumeric = parseFloat(numeroLimpo) * sinal;" & Chr(13)
    htmlCompleto = htmlCompleto & "                    acoes.push({ ticker: ticker, variacao: variacaoNumeric });" & Chr(13)
    htmlCompleto = htmlCompleto & "                }" & Chr(13)
    htmlCompleto = htmlCompleto & "            });" & Chr(13)
    htmlCompleto = htmlCompleto & "            localStorage.setItem('acoes', JSON.stringify(acoes));" & Chr(13)
    htmlCompleto = htmlCompleto & "        }" & Chr(13)
    htmlCompleto = htmlCompleto & "        window.onload = lerTabelaESalvar;" & Chr(13)
    htmlCompleto = htmlCompleto & "    </script>" & Chr(13)
    
    htmlCompleto = htmlCompleto & "</body>" & Chr(13)
    htmlCompleto = htmlCompleto & "</html>"
    
    ' Grava o arquivo final mercado.html limpo e sem bugs de sintaxe
    Dim oSimpleFileAccess As Object
    Dim oOutputStream As Object
    Dim oTextStream As Object
    oSimpleFileAccess = CreateUnoService("com.sun.star.ucb.SimpleFileAccess")
    
    If oSimpleFileAccess.exists(ConvertToUrl(sHtmlPath)) Then
        oSimpleFileAccess.kill(ConvertToUrl(sHtmlPath))
    End If
    
    oOutputStream = oSimpleFileAccess.openFileWrite(ConvertToUrl(sHtmlPath))
    oTextStream = CreateUnoService("com.sun.star.io.TextOutputStream")
    oTextStream.setOutputStream(oOutputStream)
    oTextStream.setEncoding("UTF-8")
    oTextStream.writeString(htmlCompleto)
    oTextStream.closeOutput()
End Sub

Sub SalvarHistorico

    Dim doc As Object

    Dim mercado As Object
    Dim historico As Object

    Dim ultimaLinha As Integer

    Dim coluna As Integer

    Dim codigo As String
    Dim preco As Double

    Dim numeroPregao As Integer


    doc = ThisComponent


    mercado = doc.Sheets.getByName("MERCADO")

    historico = doc.Sheets.getByName("HISTÓRICO")


    ' descobrir próxima linha vazia

    ultimaLinha = 1

    Do While historico.getCellByPosition(0, ultimaLinha).String <> ""

        ultimaLinha = ultimaLinha + 1

    Loop


    ' número do pregão

    numeroPregao = ultimaLinha


    ' escrever número do pregão

    historico.getCellByPosition(0, ultimaLinha).Value = numeroPregao


    ' percorrer empresas

    coluna = 1

    Do While mercado.getCellByPosition(0, coluna).String <> ""


        codigo = mercado.getCellByPosition(0, coluna).String

        preco = mercado.getCellByPosition(2, coluna).Value


        ' criar cabeçalho se necessário

        If historico.getCellByPosition(coluna, 0).String = "" Then

            historico.getCellByPosition(coluna, 0).String = codigo

        End If


        ' salvar preço

        historico.getCellByPosition(coluna, ultimaLinha).Value = preco


        coluna = coluna + 1

    Loop


 

End Sub
Sub ExportarGraficoLegendaLateral

    ' --- VARIÁVEIS PARA ARQUIVO UTF-8 ---

    Dim oOutputStream As Object
    Dim oTextStream As Object

    ' ---------------------

    sDataLabels = ""
    
    ' Captura as datas (Eixo X)
    iRow = 1 
    Do While oSheet.getCellByPosition(0, iRow).String <> ""
        sDataLabels = sDataLabels & "'" & oSheet.getCellByPosition(0, iRow).String & "',"
        iRow = iRow + 1
    Loop
    If Len(sDataLabels) > 0 Then sDataLabels = Left(sDataLabels, Len(sDataLabels) - 1)

    oTextStream.writeString("    <div class='container'>" & CHR(10))

    oTextStream.writeString("        <p>📊 Seleção: Clique nos nomes das empresas à direita para ocultar ou exibir as linhas no gráfico.</p>" & CHR(10))

    oTextStream.writeString("        new Chart(ctx, {" & CHR(10))

    oTextStream.writeString("            data: {" & CHR(10))

    oTextStream.writeString("                datasets: [" & CHR(10))
    
    ' Loop pelas colunas (Empresas)
    iCol = 1 
    Dim sNomeEmpresa As String
    Dim sValores As String
    Dim sCores As Variant
    Dim r As Integer
    
    ' Paleta expandida com 10 cores marcantes e diferentes para o modo escuro
    sCores = Array("#00b37e", "#f75a68", "#4e8fe7", "#ffb84d", "#9966ff", "#00d2df", "#ff79c6", "#50fa7b", "#ffb86c", "#bd93f9")

        oTextStream.writeString("                        backgroundColor: 'transparent'," & CHR(10))
        oTextStream.writeString("                        tension: 0.2," & CHR(10))
        oTextStream.writeString("                        pointRadius: 3," & CHR(10))
        oTextStream.writeString("                        pointHoverRadius: 6," & CHR(10))
        oTextStream.writeString("                        borderWidth: 2" & CHR(10))
        
        ' Remove a vírgula do último item para evitar erro de sintaxe JS, ou mantém nas outras

Sub GerarMercadoHTML()

Sub ExportarCarteirasJS()

Dim planilha As Object

Dim ultimaLinha As Long
Dim ultimaColuna As Long

Dim linha As Long
Dim coluna As Long

Dim conta As String
Dim empresa As String
Dim valorTexto As String

Dim js As String

Dim somaAcoes As Double
Dim valorFinanceiro As Double

Dim oSimpleFileAccess As Object
Dim oOutputStream As Object
Dim oTextStream As Object
Dim sCaminhoArquivo As String

' --- Nova variável para controlar as vírgulas do JS sem erros ---
Dim primeiroItem As Boolean

planilha = ThisComponent.Sheets.getByName("CARTEIRAS")

ultimaLinha = 1
Do While planilha.getCellByPosition(0, ultimaLinha).String <> ""
    ultimaLinha = ultimaLinha + 2
Loop
ultimaLinha = ultimaLinha - 2

ultimaColuna = 1
Do While planilha.getCellByPosition(ultimaColuna, 0).String <> ""
    ultimaColuna = ultimaColuna + 1
Loop
ultimaColuna = ultimaColuna - 1

js = "const carteiras = {" & Chr(10)

For linha = 1 To ultimaLinha Step 2

    conta = planilha.getCellByPosition(0, linha).String
    js = js & Chr(34) & conta & Chr(34) & ": {" & Chr(10)

    somaAcoes = 0
    primeiroItem = True ' Sinaliza que é o primeiro dado da conta atual

    For coluna = 1 To ultimaColuna

        empresa = planilha.getCellByPosition(coluna, 0).String

        If UCase(empresa) <> "VALOR EM AÇÕES" Then
            
            ' Se não for o primeiro item impresso na conta, adiciona a vírgula do item anterior
            If Not primeiroItem Then
                js = js & "," & Chr(10)
            End If
            primeiroItem = False

            valorFinanceiro = planilha.getCellByPosition(coluna, linha + 1).Value

            If UCase(empresa) = "OCIOSO" Then
                If valorFinanceiro = 0 Then
                    valorTexto = "R$ 0,00"
                Else
                    valorTexto = Format(valorFinanceiro, "R$ #,##0.00")
                End If
                js = js & "    " & Chr(34) & empresa & Chr(34) & ": " & Chr(34) & valorTexto & Chr(34)

            ElseIf UCase(empresa) = "TOTAL" Then
                If valorFinanceiro = 0 Then
                    valorTexto = "R$ 0,00"
                Else
                    valorTexto = Format(valorFinanceiro, "R$ #,##0.00")
                End If
                js = js & "    " & Chr(34) & empresa & Chr(34) & ": " & Chr(34) & valorTexto & Chr(34)

            Else
                somaAcoes = somaAcoes + valorFinanceiro
                valorTexto = Replace(CStr(planilha.getCellByPosition(coluna, linha).Value), ",", ".")
                js = js & "    " & Chr(34) & empresa & Chr(34) & ": " & valorTexto
            End If

        End If

    Next coluna

    ' --- INSERE O "VALOR EM AÇÕES" CALCULADO COM A VÍRGULA CORRETA ---
    If somaAcoes = 0 Then
        valorTexto = "R$ 0,00"
    Else
        valorTexto = Format(somaAcoes, "R$ #,##0.00")
    End If

    ' Se adicionamos algum item antes, coloca a vírgula antes de injetar o VALOR EM AÇÕES
    If Not primeiroItem Then
        js = js & "," & Chr(10)
    End If
    
    js = js & "    " & Chr(34) & "VALOR EM AÇÕES" & Chr(34) & ": " & Chr(34) & valorTexto & Chr(34) & Chr(10)
    js = js & "}"

    If linha < ultimaLinha Then
        js = js & ","
    End If
    js = js & Chr(10)

Next linha

js = js & "};"

' --- GRAVAÇÃO EM UTF-8 PURO ---
sCaminhoArquivo = ConvertToURL("C:\Users\victo\OneDrive\Desktop\BOLSA\carteiras.js")

oSimpleFileAccess = CreateUnoService("com.sun.star.ucb.SimpleFileAccess")

If oSimpleFileAccess.exists(sCaminhoArquivo) Then
    oSimpleFileAccess.kill(sCaminhoArquivo)
End If

oOutputStream = oSimpleFileAccess.openFileWrite(sCaminhoArquivo)
oTextStream = CreateUnoService("com.sun.star.io.TextOutputStream")
oTextStream.setOutputStream(oOutputStream)
oTextStream.setEncoding("UTF-8")

oTextStream.writeString(js)
oTextStream.closeOutput()



End Sub
