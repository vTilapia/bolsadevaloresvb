Sub ExecutarOrdens

Dim ordens
Dim alunos
Dim carteira
Dim mercado

Dim i As Integer
Dim linhaAluno As Integer
Dim linhaMercado As Integer
Dim colunaEmpresa As Integer

Dim conta As String
Dim codigo As String
Dim tipo As String
Dim qtd As Double

Dim saldo As Double
Dim preco As Double
Dim valor As Double
Dim acoes As Double

ordens=ThisComponent.Sheets.getByName("ORDENS")
alunos=ThisComponent.Sheets.getByName("INVESTIDORES")
carteira=ThisComponent.Sheets.getByName("CARTEIRAS")
mercado=ThisComponent.Sheets.getByName("BOLSA DE VALORES")


For i=1 To 30

conta=ordens.getCellByPosition(1,i).String
tipo=UCase(ordens.getCellByPosition(3,i).String)
qtd=Val(Trim(ordens.getCellByPosition(4, i).String))
codigo=ordens.getCellByPosition(2,i).String

If conta<>"" Then

linhaAluno=-1

For j=1 To 100

If carteira.getCellByPosition(0,j).String=conta Then
linhaAluno=j
Exit For
End If

Next j


If linhaAluno=-1 Then
MsgBox("Conta "& conta &" não encontrada")
GoTo ProximaOrdem
End If


linhaMercado=-1

For k=1 To 100

If mercado.getCellByPosition(1,k).String=codigo Then
linhaMercado=k
Exit For
End If

Next k


If linhaMercado=-1 Then
MsgBox("Empresa "& codigo &" não encontrada")
GoTo ProximaOrdem
End If


colunaEmpresa=-1

For j=1 To 50

If carteira.getCellByPosition(j,0).String=codigo Then
colunaEmpresa=j
Exit For
End If

Next j


saldo=alunos.getCellByPosition(5,linhaAluno).Value

preco=mercado.getCellByPosition(5,linhaMercado).Value

valor=preco*qtd

acoes=carteira.getCellByPosition(colunaEmpresa,linhaAluno).Value


If tipo="COMPRA" Then

If saldo>=valor Then

saldo=saldo-valor
acoes=acoes+qtd

Else

MsgBox("Saldo insuficiente: "&conta)

End If

ElseIf tipo="VENDA" Then

If acoes>=qtd Then

saldo=saldo+valor
acoes=acoes-qtd

Else

MsgBox("Ações insuficientes: "&conta)

End If

End If


alunos.getCellByPosition(5,linhaAluno).Value=saldo

carteira.getCellByPosition(colunaEmpresa,linhaAluno).Value=acoes


ProximaOrdem:

End If

Next i

MsgBox("Ordens executadas!")

End Sub

Sub SalvarHistorico

    Dim doc As Object

    Dim mercado As Object
    Dim historico As Object

    Dim ultimaLinha As Integer

    Dim coluna As Integer

    Dim codigo As String
    Dim preco As Double

    Dim numeroPregao As Integer


    doc = ThisComponent


    mercado = doc.Sheets.getByName("MERCADO")

    historico = doc.Sheets.getByName("HISTÓRICO")


    ' descobrir próxima linha vazia

    ultimaLinha = 1

    Do While historico.getCellByPosition(0, ultimaLinha).String <> ""

        ultimaLinha = ultimaLinha + 1

    Loop


    ' número do pregão

    numeroPregao = ultimaLinha


    ' escrever número do pregão

    historico.getCellByPosition(0, ultimaLinha).Value = numeroPregao


    ' percorrer empresas

    coluna = 1

    Do While mercado.getCellByPosition(0, coluna).String <> ""


        codigo = mercado.getCellByPosition(0, coluna).String

        preco = mercado.getCellByPosition(2, coluna).Value


        ' criar cabeçalho se necessário

        If historico.getCellByPosition(coluna, 0).String = "" Then

            historico.getCellByPosition(coluna, 0).String = codigo

        End If


        ' salvar preço

        historico.getCellByPosition(coluna, ultimaLinha).Value = preco


        coluna = coluna + 1

    Loop


    MsgBox "Histórico salvo!"

End Sub

Sub GerarMercadoHTML()

    Dim doc As Object
    Dim planilha As Object

    Dim linha As Integer
    Dim ultimaLinha As Integer

    Dim codigo As String
    Dim empresa As String
    Dim preco As Double
    Dim variacao As Double

    Dim html As String

    doc = ThisComponent

    planilha = doc.Sheets.getByName("MERCADO")

    ultimaLinha = 1

    Do While planilha.getCellByPosition(0, ultimaLinha).String <> ""

        ultimaLinha = ultimaLinha + 1

    Loop

    html = "<!DOCTYPE html>"
    html = html & "<html lang='pt-BR'>"
    html = html & "<head>"
    html = html & "<meta charset='UTF-8'>"
    html = html & "<meta name='viewport' content='width=device-width, initial-scale=1.0'>"

    html = html & "<style>"

    html = html & "body{margin:0;padding:10px;background:#111827;font-family:Arial,sans-serif;color:white;}"

    html = html & "h1{text-align:center;font-size:22px;margin-bottom:15px;}"

    html = html & ".container{overflow-x:auto;}"

    html = html & "table{width:100%;border-collapse:collapse;background:#1f2937;border-radius:12px;overflow:hidden;font-size:14px;}"

    html = html & "th{background:#f4b400;color:black;padding:10px 6px;}"

    html = html & "td{padding:10px 6px;text-align:center;border-bottom:1px solid #374151;color:white;}"

    html = html & ".alta{color:#22c55e;font-weight:bold;}"

    html = html & "@media screen and (max-width:600px){"

    html = html & "h1{font-size:18px;}"

    html = html & "table{font-size:12px;}"

    html = html & "th,td{padding:8px 4px;}"

    html = html & "}"

    html = html & "</style>"

    html = html & "</head>"

    html = html & "<body>"

    html = html & "<h1>📈 Bolsa Escolar</h1>"

    html = html & "<div class='container'>"

    html = html & "<table>"

    html = html & "<tr>"
    html = html & "<th>Código</th>"
    html = html & "<th>Empresa</th>"
    html = html & "<th>Preço</th>"
    html = html & "<th>Variação</th>"
    html = html & "</tr>"


    For linha = 1 To ultimaLinha

        codigo = planilha.getCellByPosition(0, linha).String
        empresa = planilha.getCellByPosition(1, linha).String

            html = html & "<td class='baixa'>&#9660; " & Replace(Format(Abs(variacao), "0.00"), ",", ".") & "%</td>"

        End If

        html = html & "</tr>"

    Next linha


    html = html & "</table>"

    html = html & "</div>"

    html = html & "</body>"

    html = html & "</html>"


    Dim arquivo As Integer

    arquivo = FreeFile

    Open "C:\Users\victo\OneDrive\Desktop\BOLSA\mercado.html" For Output As #arquivo

    Print #arquivo, html

    Close #arquivo

    MsgBox "mercado.html criado com sucesso!"

End Sub

Sub ImportarOrdensCSV()

    Dim planilha
    Dim arquivo As Integer
    Dim linhaTexto As String
    Dim campos()
    Dim linhaDestino As Integer
    Dim coluna As Integer

    planilha = ThisComponent.Sheets.getByName("ORDENS")

    planilha.getCellRangeByPosition(0,0,20,1000).clearContents(1023)

    arquivo = FreeFile

    Open "C:\Users\victo\OneDrive\Desktop\BOLSA\ordens.csv" For Input As #arquivo

    linhaDestino = 0

    Do Until EOF(arquivo)

        Line Input #arquivo, linhaTexto

        campos = Split(linhaTexto, ",")

        For coluna = 0 To UBound(campos)

            planilha.getCellByPosition(coluna, linhaDestino).String = Trim(campos(coluna))

        Next coluna

        linhaDestino = linhaDestino + 1

    Loop

    Close #arquivo

    MsgBox "Ordens importadas!"

End Sub

Sub ExportarCarteirasJS()


Dim doc As Object
Dim planilha As Object

Dim ultimaLinha As Long
Dim ultimaColuna As Long

Dim linha As Long
Dim coluna As Long

Dim conta As String
Dim empresa As String
Dim quantidade As Double

Dim js As String

Dim arquivo As Integer

doc = ThisComponent

planilha = doc.Sheets.getByName("CARTEIRAS")

ultimaLinha = 1

Do While planilha.getCellByPosition(0, ultimaLinha).String <> ""
    ultimaLinha = ultimaLinha + 2
Loop

ultimaLinha = ultimaLinha - 2

ultimaColuna = 1

Do While planilha.getCellByPosition(ultimaColuna, 0).String <> ""
    ultimaColuna = ultimaColuna + 1
Loop

ultimaColuna = ultimaColuna - 1

js = "const carteiras = {" & Chr(10)

For linha = 1 To ultimaLinha Step 2

    conta = planilha.getCellByPosition(0, linha).String

    js = js & "  """ & conta & """: {" & Chr(10)

    For coluna = 1 To ultimaColuna

        empresa = planilha.getCellByPosition(coluna, 0).String

        quantidade = planilha.getCellByPosition(coluna, linha).Value

        js = js & "    """ & empresa & """: " & Replace(CStr(quantidade), ",", ".")

        If coluna < ultimaColuna Then
            js = js & ","
        End If

        js = js & Chr(10)

    Next coluna

    js = js & "  }"

    If linha < ultimaLinha Then
        js = js & ","
    End If

    js = js & Chr(10)

Next linha

js = js & "};"

arquivo = FreeFile

Open "C:\Users\victo\OneDrive\Desktop\BOLSA\carteiras.js" For Output As #arquivo

Print #arquivo, js

Close #arquivo

MsgBox "carteiras.js criado com sucesso!"


End Sub
