Sub ExportarCarteirasJS()

```
Dim planilha As Object

Dim ultimaLinha As Long
Dim ultimaColuna As Long

Dim linha As Long
Dim coluna As Long

Dim conta As String
Dim empresa As String
Dim valorTexto As String

Dim js As String

Dim arquivo As Integer

planilha = ThisComponent.Sheets.getByName("CARTEIRAS")

ultimaLinha = 1

Do While planilha.getCellByPosition(0, ultimaLinha).String <> ""

    ultimaLinha = ultimaLinha + 2

Loop

ultimaLinha = ultimaLinha - 2

ultimaColuna = 1

Do While planilha.getCellByPosition(ultimaColuna, 0).String <> ""

    ultimaColuna = ultimaColuna + 1

Loop

ultimaColuna = ultimaColuna - 1

js = "const carteiras = {" & Chr(10)

For linha = 1 To ultimaLinha Step 2

    conta = planilha.getCellByPosition(0, linha).String

    js = js & Chr(34) & conta & Chr(34) & ": {" & Chr(10)

    For coluna = 1 To ultimaColuna

        empresa = planilha.getCellByPosition(coluna, 0).String

        valorTexto = Replace(CStr(planilha.getCellByPosition(coluna, linha).Value), ",", ".")

        js = js & Chr(34) & empresa & Chr(34) & ": " & valorTexto

        If coluna < ultimaColuna Then
            js = js & ","
        End If

        js = js & Chr(10)

    Next coluna

    js = js & "}"

    If linha < ultimaLinha Then
        js = js & ","
    End If

    js = js & Chr(10)

Next linha

js = js & "};"

arquivo = FreeFile

Open "C:\Users\victo\OneDrive\Desktop\BOLSA\carteiras.js" For Output As #arquivo

Print #arquivo, js

Close #arquivo

MsgBox "carteiras.js criado com sucesso!"
```

End Sub
